//------------------------------------------------------------------------------
actor HolyTail
{
	Radius 1
	Height 1
	RenderStyle Translucent
	Alpha 0.6
	+NoBlockmap
	+NoGravity
	+DropOff
	+NoClip
	+NoTeleport

	action native A_CHolyTail();

	states
	{
	Spawn:
		SPIR C 1 A_CHolyTail
		Loop
	TailTrail:
		SPIR D -1
		Stop
	}
}

//------------------------------------------------------------------------------
actor HolyTailTrail : HolyTail
{
	states
	{
	Spawn:
		Goto TailTrail
	}
}
