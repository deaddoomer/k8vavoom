actor ClericPlayer : PlayerPawn
{
	Health 100
	Radius 16
	Height 64
	PainChance 255
	RadiusDamageFactor 0.25
	PainSound "PlayerClericPain"
	Player.ViewHeight 48
	Player.ColorRange 146, 163
	Player.HexenArmor 10, 10, 25, 5, 20
	Player.JumpZ 9.5
	Player.SoundClass "cleric"
	Player.DisplayName "Cleric"
	Player.HealRadiusType "Health"
	Player.InvulnerabilityMode "Ghost"
	Player.SpawnClass "Cleric"
	Player.ScoreIcon "clerface"
	Player.StartItem "CWeapMace"
	states
	{
	Spawn:
		CLER A -1
		Stop
	See:
		CLER ABCD 4
		Loop
	Melee:
	Missile:
		CLER EFG 6
		Goto Spawn
	Pain:
		CLER H 4
		CLER H 4 A_Pain
		Goto Spawn
	Death:
		CLER I 6
		CLER K 6 A_PlayerScream
		CLER LL 6
		CLER M 6 A_NoBlocking
		CLER NOP 6
		CLER Q -1
		Stop
	XDeath:
		CLER R 5 A_PlayerScream
		CLER S 5
		CLER T 5 A_NoBlocking
		CLER UVWXYZ 5
		CLER [ -1
		Stop
	Ice:
		CLER "\" 5 A_FreezeDeath
		CLER "\" 1 A_FreezeDeathChunks
		Wait
	Burn:
		FDTH C 5 Bright A_PlaySound("*burndeath")
		FDTH D 4 Bright
		FDTH G 5 Bright
		FDTH H 4 Bright A_PlayerScream
		FDTH I 5 Bright
		FDTH J 4 Bright
		FDTH K 5 Bright
		FDTH L 4 Bright
		FDTH M 5 Bright
		FDTH N 4 Bright
		FDTH O 5 Bright
		FDTH P 4 Bright
		FDTH Q 5 Bright
		FDTH R 4 Bright
		FDTH S 5 Bright A_NoBlocking
		FDTH T 4 Bright
		FDTH U 5 Bright
		FDTH V 4 Bright
		ACLO E 35 A_CheckPlayerDone
		Wait
		ACLO E 8
		Stop
	}
}
