// this is main blood actor; it is spawned by the engine
#ifndef TRANSIENT
Actor K8Gore_Blood$(Color) : Blood$(Color) /*: K8Gore_BloodBase$(TransientSfx)*/ Replaces Blood$(Color) {
#else
Actor K8Gore_Blood$(Color) /*: K8Gore_BloodBase$(TransientSfx)*/ {
#endif
  $(Translation)
  Decal K8GoreBloodSplat$(Color)
  game Doom
  +FORCEXYBILLBOARD
  +NOGRAVITY
  +THRUACTORS
  +CLIENTSIDEONLY
  -ALLOWPARTICLES
  Scale 0.5
  Speed 0

  var int user_spray;

  States {
    Spawn:
      TNT1 A 0
      TNT1 AAAA 0 {
        if (GetCVar("k8GoreOpt_BloodAmount") >= 3) {
          A_CustomMissile("K8Gore_FlyingBloodFake$(Color)", 0, 0, random(0, 360), 2, random(0, 90));
        }
      }

#ifndef TRANSIENT
      // normal
      TNT1 AAAA 0 {
        if (user_spray && GetCVar("k8GoreOpt_BloodAmount") >= 3) {
          A_CustomMissile("K8Gore_FlyingBloodFake$(Color)", 0, 0, random(0, 360), 2, random(0, 90));
        }
      }
#endif

      TNT1 A 0 {
/*
#ifndef TRANSIENT
        // normal
        if (GetCVar("k8GoreOpt_BloodAmount") >= 3) A_CustomMissile("K8Gore_FlyingBloodLevel3$(Color)", 7, 0, random(0, 360), 2, random(40, 60));
#endif
*/
        // all
        if (GetCVar("k8GoreOpt_BloodAmount") >= 3) A_CustomMissile("K8Gore_FlyingBloodLevel3$(Color)", 7, 0, random(0, 360), 2, random(40, 60));

        // all
        if (GetCVar("k8GoreOpt_BloodAmount") >= 2) A_CustomMissile("K8Gore_FlyingBlood$(Color)", 7, 0, random(0, 360), 2, random(20, 40));
#ifndef TRANSIENT
        // normal
        if (user_spray && GetCVar("k8GoreOpt_BloodAmount") >= 1) A_CustomMissile("K8Gore_BloodmistSmall$(Color)", 0, 0, random(0, 360), 2, random(10, 40));
#else
        // transient
        if (GetCVar("k8GoreOpt_BloodAmount") >= 1) A_CustomMissile("K8Gore_FlyingBlood$(Color)", 7, 0, random(0, 360), 2, random(20, 40));
#endif
        // all
        if (GetCVar("k8GoreOpt_BloodAmount") >= 0) A_CustomMissile("K8Gore_BloodHit$(Color)", 0, 0, random(0, 360), 2, random(0, 90));
      }
      TNT1 AAAA 0 A_CustomMissile("K8Gore_FlyingBloodTrail$(Color)", 0, 0, random (0, 360), 2, random (0, 180))
      Stop

    Spray:
      TNT1 A 0 A_SetUserVar("user_spray", 1)
      Goto Spawn
  }
}

// flying blood; will spawn blood spots and ceiling blood
Actor K8Gore_FlyingBlood$(Color) /*: K8Gore_BloodBase$(TransientSfx)*/ {
  $(Translation)
  game Doom
  scale 0.5
  speed 4
  health 1
  radius 8
  height 1
  Gravity 0.4 //0.8
  damage 0
  //Renderstyle Translucent
  //Alpha 0.95
  DamageType Blood
  Decal K8GoreBloodSuper$(Color)
  +MISSILE
  +CLIENTSIDEONLY
  +NOTELEPORT
  +NOBLOCKMAP
  +THRUACTORS
  +BLOODLESSIMPACT
  +FORCEXYBILLBOARD
  +NODAMAGETHRUST
  +MOVEWITHSECTOR
  +CORPSE
  -DONTSPLASH
  -NOGRAVITY
  Mass 1

  States {
    Spawn:
      TNT1 A 0
      TNT1 A 0 ThrustThingZ(0, random(2, 4)*4, 0, 1)
      B8OD AB 2
      TNT1 A 0 A_JumpIf(waterlevel > 1, "Splash")
      TNT1 A 0 A_Jump(256, "Spawn1", "Spawn2", "Spawn3")

    Spawn1:
      B8OD C 0
      Goto MainWorker
    Spawn2:
      B8OD D 0
      Goto MainWorker
    Spawn3:
      B8OD E 0
      Goto MainWorker

    MainWorker:
      "####" "#" 0 A_JumpIf(waterlevel > 1, "Splash")
    MainLoop:
#ifndef TRANSIENT
      "####" "#" 1
#else
      "####" "#" 1 A_FadeOut(0.01)
#endif
      Loop

    Splash:
      "####" "#" 1 A_FadeOut(0.01)
      Loop

    // hits something
    Death:
      TNT1 A 0
      TNT1 A 0 { if (GetCVar("k8GoreOpt_BloodDripSound") > 0) A_PlaySound("k8gore/blooddrop2"); }
      "####" "#" 0 A_JumpIf(waterlevel > 1, "DeathDone")
      TNT1 A 0 A_SpawnItem("K8Gore_BloodSpot$(Color)", 0, 1)
      //TNT1 A 0 A_CustomMissile("K8Gore_CeilBloodChecker$(Color)", -15, 0, 90, 2, 90)
      TNT1 A 0 A_SpawnItemEx("K8Gore_DetectCeilBloodSimpler$(Color)",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION,0)
    DeathDone:
      TNT1 A 1
      Stop
   }
}

// this is floor blood spot (it may be spawned in the air, though)
Actor K8Gore_BloodSpot$(Color) : K8Gore_BloodBase$(TransientSfx) {
  $(Translation)
  game Doom
  radius 2
  height 2
  Health 300
  +NOTELEPORT
  +CLIENTSIDEONLY
  +FORCEXYBILLBOARD
  +MOVEWITHSECTOR
  +FLOORCLIP
  +NOTARGET
  -SHOOTABLE
  +NOBLOOD
  -DONTSPLASH
  -SOLID
  +THRUACTORS
  -NOGRAVITY
  -NoLiftDrop
  +NOBLOCKMAP
  //+NoInteraction
  //Renderstyle Translucent
#ifdef TRANSIENT
  //!Alpha 0.6
  //Alpha 0.7
#else
  //Alpha 0.9
#endif
  Mass 99999
  Scale 0.5
  Decal K8GoreBloodSplat$(Color)
  DamageFactor "TeleportRemover", 100.0

  States {
    Spawn:
      TNT1 A 0
      TNT1 A 0 A_JumpIfCVarFalse("r_models", "Death") // no models -- die immediately
      // this has no sense, as blood is not in blockmap
      TNT1 A 0 A_LayToFloorWithYaw(Random(0, 360))
      //TNT1 A 0 A_SetRollExt(0)
      //TNT1 A 0 A_SetYawExt(Random(0, 360))
      //TNT1 A 0 A_SetPitchExt(0)
      TNT1 A 0 A_SetScaleXExt(Random(3, 9)/10.0)
      TNT1 A 0 A_JumpIf(waterlevel > 1, "SplashWater")

    SpawnNormally:
      TNT1 A 0 A_Jump(256, "SpawnAFrameOnly", "SpawnBFrameOnly", "SpawnCFrameOnly", "SpawnDFrameOnly")
    SpawnAFrameOnly:
      BPDL A 0
      Goto DoAfterSpawn
    SpawnBFrameOnly:
      BPDL B 0
      Goto DoAfterSpawn
    SpawnCFrameOnly:
      BPDL C 0
      Goto DoAfterSpawn
    SpawnDFrameOnly:
      BPDL D 0
      Goto DoAfterSpawn

#ifdef TRANSIENT
    DoAfterSpawn:
      "####" "#" 35 A_CheckFloor("ItLanded")
      Loop
    ItLanded:
      "####" "#" 0 A_ChangeFlag("NoInteraction", true)
      "####" "#" 0 A_k8JumpIfLiquidFloor("ItLandedOnWater")
      "####" "#" random(100, 450)
    FadeOutLoop:
      "####" "#" random(64, 90)
      "####" "#" 0 A_FadeOut(0.01)
      Loop
#else
    DoAfterSpawn:
      "####" "#" 16 A_CheckFloor("ItLanded")
      Loop
    ItLanded:
      "####" "#" 0 A_ChangeFlag("NoInteraction", true)
      "####" "#" 0 A_k8JumpIfLiquidFloor("ItLandedOnWater")
    EndlessLoop:
      "####" "#" 800
      // it still may disappear after some time
      "####" "#" 0 A_Jump(16, "NormalFadeOut")
      Loop
    NormalFadeOut:
      "####" "#" random(64, 90)
      "####" "#" 0 A_FadeOut(0.01)
      Loop
#endif


    SplashWater:
      TNT1 A 0 A_FadeOut(0.3)
      TNT1 A 0 A_Jump(256, "SpawnAFrameOnlyWater", "SpawnBFrameOnlyWater", "SpawnCFrameOnlyWater", "SpawnDFrameOnlyWater")
    SpawnAFrameOnlyWater:
      BPDL A 0
      Goto WaterFadeMain
    SpawnBFrameOnlyWater:
      BPDL B 0
      Goto WaterFadeMain
    SpawnCFrameOnlyWater:
      BPDL C 0
      Goto WaterFadeMain
    SpawnDFrameOnlyWater:
      BPDL D 0
      Goto WaterFadeMain

    WaterFadeMain:
      "####" "#" 16 A_CheckFloor("ItLandedOnWater")
      Loop
    ItLandedOnWater:
      "####" "#" 0 A_ChangeFlag("NoInteraction", true)
      // stay for some time
      "####" "#" random(50, 100)
      // and fade out
    WaterFadeLoop:
      "####" "#" random(9, 24)
      "####" "#" 0 A_FadeOut(0.05)
      Loop


    Death:
      TNT1 A 0 A_NoBlocking
      Stop
  }
}
