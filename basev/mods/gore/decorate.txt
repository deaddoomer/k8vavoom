// this is main blood actor; it is spawned by the engine
// Args[0] is damage (0 means "unknown")
// Tracer is damaged actor (can be empty)
Actor K8Gore_Blood : Blood /*: K8Gore_BloodBase*/ Replaces Blood {
  
  Decal None
  //+Decals2SPass
  game Doom
  +FORCEXYBILLBOARD
  +NOGRAVITY
  +THRUACTORS
  +CLIENTSIDEONLY
  +NoTeleport
  -ALLOWPARTICLES
  +CopyBloodTranslation
  +k8DetachFromServer
  +k8AllowSimpleTick
  Scale 0.5
  Speed 0

  var int user_spray;

  States {
    Spawn:
      TNT1 A 0
      /*TNT1 A 0 A_k8ConLog("blood: Spawn: 000")*/
      TNT1 AAAA 0 {
        if (Args[0] > 20) A_SetUserVar("user_spray", 1);
        if (GetCVar("k8GoreOpt_BloodAmount") >= 3) {
          A_CustomMissile("K8Gore_FlyingBloodFake", 0, 0, random(0, 360), CMF_AIMDIRECTION, random(0, 90));
        }
      }

      // normal
      TNT1 AAAA 0 {
        if (user_spray || GetCVar("k8GoreOpt_BloodAmount") >= 3) {
          A_CustomMissile("K8Gore_FlyingBloodFake", 0, 0, random(0, 360), CMF_AIMDIRECTION, random(0, 90));
        }
      }

      TNT1 A 0 {
        // all
        if (GetCVar("k8GoreOpt_BloodAmount") >= 3) A_CustomMissile("K8Gore_FlyingBloodLevel3", 7, 0, random(0, 360), CMF_AIMDIRECTION, random(40, 60));

        // all
        if (GetCVar("k8GoreOpt_BloodAmount") >= 2) A_CustomMissile("K8Gore_FlyingBlood", 7, 0, random(0, 360), CMF_AIMDIRECTION, random(20, 40));
        // normal
        // all
        if (GetCVar("k8GoreOpt_BloodAmount") >= 0) A_CustomMissile("K8Gore_BloodHit", 0, 0, random(0, 360), CMF_AIMDIRECTION, random(0, 90));
      }
      TNT1 AAAA 0 A_CustomMissile("K8Gore_FlyingBloodTrail", 0, 0, random (0, 360), CMF_AIMDIRECTION, random (0, 180))
      /*TNT1 A 0 A_k8ConLog("blood: Spawn Done")*/
      TNT1 A 0 A_JumpIf(IsListenServer, "InListenServer")
      Stop

    InListenServer:
      TNT1 A 0 A_ChangeFlag("k8ServerSideOnly", false)
      /*TNT1 A 0 A_k8ConLog("blood: listen server")*/
      /* fallback */

    /* for both server modes if `SpawnInListenServer` is absent */
    SpawnInNetServer:
      TNT1 A 0
      /* for non-dedicated server we still want blood */
      /*TNT1 A 0 A_k8ConLog("blood: dedicated server")*/
      TNT1 A 8
      Stop

    SpawnInListenServer:
      TNT1 A 0
      TNT1 A 0 A_ChangeFlag("k8ServerSideOnly", true)
      /*TNT1 A 0 A_k8ConLog("blood: listen server spawn")*/
      Goto Spawn

    NetDetachedOnClient:
      TNT1 A 0
      /*TNT1 A 0 A_k8ConLog("blood: detached")*/
      Goto Spawn

    /*
    Spray:
      TNT1 A 0 A_SetUserVar("user_spray", 1)
      Goto Spawn
    */
  }
}

      "####" "#" 1 A_FadeOut(0.01)
      Loop

      TNT1 A 0
      //k8: don't make any sound, it is not noticeable anyway
      //TNT1 A 0 { if (GetCVar("k8GoreOpt_BloodDripSound") > 0) A_PlaySound_NoNet("k8gore/blooddrop2"); }
      //"####" "#" 0 A_JumpIf(waterlevel > 1, "DeathDone")
      //TNT1 A 0 A_SpawnItem("K8Gore_BloodSpot", 0, 1, false, true)
      //TNT1 A 0 A_CustomMissile("K8Gore_CeilBloodChecker", -15, 0, 90, CMF_AIMDIRECTION, 90)

            if (GetCVar("k8GoreOpt_FloorBlood")) {
              A_SpawnItem("K8Gore_BloodSpot", 0, 1, false, true);
            }
            // spawn ceiling blood spot
            if (GetCVar("k8GoreOpt_CeilBlood") && GetCVar("k8GoreOpt_BloodAmount") >= 2) {
              if (ceilingz-z < random(118, 132)) {
                if ((ceilingz-z <= 64 && random(0, 100) > 20) ||
                    (ceilingz-z > 64 && ceilingz-z < 80 && random(0, 100) > 40) ||
                    (ceilingz-z >= 80 && random(0, 100) > 60))

                  A_SpawnItem("K8Gore_CeilBloodSpot", 0/*distance*/, 0/*zheight*/, 0/*useammo*/, true/*transferblood*/);
                }
              }
            }
          }
        }
      Stop
    DeathDone:
      TNT1 A 1
      Stop
   }
}

Actor K8Gore_BloodSpot : K8Gore_BloodBase {
  LimitWithSubCvar "k8GoreOpt_MaxBloodSplatEntities"
  
  game Doom
  radius 0
  height 0
  Health 300
  +NOTELEPORT
  +CLIENTSIDEONLY
  +MOVEWITHSECTOR
  +FLOORCLIP
  +NOTARGET
  -SHOOTABLE
  +NOBLOOD
  -DONTSPLASH
  -SOLID
  +THRUACTORS
  -NOGRAVITY
  -NoLiftDrop
  +NOBLOCKMAP
  +OrientedOfs
  +CopyBloodTranslation
  +k8DetachFromServer
  +k8ServerSideOnly
  +k8AllowSimpleTick
  Renderstyle Dark
  Alpha 0.9
  Mass 99999
  Scale 0.5
  Decal None
  -Decals2SPass
  //Decal K8GoreBloodSplat
  //+Decals2SPass

  States {
    Spawn:
      TNT1 A 0
      TNT1 A 0 {
        A_LayToFloorWithYawSprite(Random(0, 360));
        A_SetScaleXExt(Random(3, 9)/10.0);
      }
      TNT1 A 0 A_JumpIf(waterlevel > 1, "SplashWater")

    SpawnNormally:
      TNT1 A 0 A_Jump(256, "SpawnAFrameOnly", "SpawnBFrameOnly", "SpawnCFrameOnly", "SpawnDFrameOnly")
    SpawnAFrameOnly:
      BPDL A 0
      Goto DoAfterSpawn
    SpawnBFrameOnly:
      BPDL B 0
      Goto DoAfterSpawn
    SpawnCFrameOnly:
      BPDL C 0
      Goto DoAfterSpawn
    SpawnDFrameOnly:
      BPDL D 0
      Goto DoAfterSpawn

    DoAfterSpawn:
      "####" "#" 16 A_CheckFloor("ItLanded")
      Loop
    ItLanded:
      "####" "#" 0
        {
          A_ChangeVelocity(0, 0, 0, CVF_REPLACE);
          A_ChangeFlag("NoInteraction", true);
          if (!A_k8JumpIfOnFloorSky("Death")) {
            A_k8JumpIfLiquidFloor("ItLandedOnWater");
          }
        }
    EndlessLoop:
      // become non-scripted; from 22 secs to 17 minutes
      "####" "#" -1 A_BecomeK8VavoomInternalNoTickGrav(frandom(800, 12600)/35.0, frandom(64, 90)/35.0/10.0)
      /*
      "####" "#" 800
      // it still may disappear after some time
      "####" "#" 0 A_Jump(16, "NormalFadeOut")
      */
      Loop
    NormalFadeOut:
      "####" "#" random(64, 90)
      "####" "#" 0 A_FadeOut(0.01)
      Loop

    SplashWater:
      TNT1 A 0 A_FadeOut(0.3)
      TNT1 A 0 A_Jump(256, "SpawnAFrameOnlyWater", "SpawnBFrameOnlyWater", "SpawnCFrameOnlyWater", "SpawnDFrameOnlyWater")
    SpawnAFrameOnlyWater:
      BPDL A 0
      Goto WaterFadeMain
    SpawnBFrameOnlyWater:
      BPDL B 0
      Goto WaterFadeMain
    SpawnCFrameOnlyWater:
      BPDL C 0
      Goto WaterFadeMain
    SpawnDFrameOnlyWater:
      BPDL D 0
      Goto WaterFadeMain

    WaterFadeMain:
      "####" "#" 16 A_CheckFloor("ItLandedOnWater")
      Loop
    ItLandedOnWater:
      "####" "#" 0
        {
          A_ChangeVelocity(0, 0, 0, CVF_REPLACE);
          A_ChangeFlag("NoInteraction", true);
        }
      // stay for some time
      "####" "#" random(50, 100)
      // and fade out
    WaterFadeLoop:
      "####" "#" random(2, 4)
      "####" "#" 0 A_FadeOut(0.01)
      Loop


    Death:
      TNT1 A 0 A_NoBlocking
      Stop
  }
}
