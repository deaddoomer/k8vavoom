// ////////////////////////////////////////////////////////////////////////// //
// this is main blood actor; it is spawned by the engine
// Args[0] is damage (0 means "unknown")
// Tracer is damaged actor (can be empty)
//
// for normal blood, the engine will spawn:
//   Blood[Color]
//   BloodSplatter[Color]
//
// for gore blood, the engine will spawn:
//   K8Gore_Blood[Color]
//   K8Gore_BloodSplatter[Color]
//
#ifndef TRANSIENT
Actor K8Gore_Blood$(Color) : Blood$(Color) /*: K8Gore_BloodBase$(TransientSfx)*/ /*Replaces Blood$(Color)*/ {
#else
Actor K8Gore_Blood$(Color) /*: K8Gore_BloodBase$(TransientSfx)*/ {
#endif
  $(Translation)
#ifndef TRANSIENT
  Decal None
  +NOSECTOR
  +NOINTERACTION
#else
  //old:Decal K8GoreBloodSplat$(Color)
  //new start
  Decal K8GoreBloodSplat$(Color)
  +NOSECTOR
  +NOINTERACTION
  //new end
#endif
  //+Decals2SPass
  game Doom
  +FORCEXYBILLBOARD
  +NOGRAVITY
  +THRUACTORS
  +NOBLOCKMAP
  +CLIENTSIDEONLY
  +NoTeleport
  -ALLOWPARTICLES
  +CopyBloodTranslation
  +k8DetachFromServer
  +k8AllowSimpleTick
  Scale 0.5
  Speed 0
  Radius 3
  Height 3

  var int user_spray;

  States {
    Spawn:
      TNT1 A 0
      /*TNT1 A 0 A_k8ConLog("blood: Spawn: 000")*/
      TNT1 AAAA 0 {
        if (Args[0] > 20) A_SetUserVar("user_spray", 1);
        if (GetCVar("k8GoreOpt_BloodAmount") >= 3) {
          A_CustomMissile("K8Gore_FlyingBloodFake$(Color)", 0, 0, random(0, 360), CMF_AIMDIRECTION, random(0, 90));
        }
      }

#ifndef TRANSIENT
      // normal
      TNT1 AAAA 0 {
        if (user_spray || GetCVar("k8GoreOpt_BloodAmount") >= 3) {
          A_CustomMissile("K8Gore_FlyingBloodFake$(Color)", 0, 0, random(0, 360), CMF_AIMDIRECTION, random(0, 90));
        }
      }
#endif

      TNT1 A 0 {
        // all
        if (GetCVar("k8GoreOpt_BloodAmount") >= 3) A_CustomMissile("K8Gore_FlyingBloodLevel3$(Color)", 7, 0, random(0, 360), CMF_AIMDIRECTION, random(40, 60));

        // all
        if (GetCVar("k8GoreOpt_BloodAmount") >= 2) A_CustomMissile("K8Gore_FlyingBlood$(Color)", 7, 0, random(0, 360), CMF_AIMDIRECTION, random(20, 40));
#ifndef TRANSIENT
        // normal
#else
        // transient
        if (GetCVar("k8GoreOpt_BloodAmount") >= 1) A_CustomMissile("K8Gore_FlyingBlood$(Color)", 7, 0, random(0, 360), CMF_AIMDIRECTION, random(20, 40));
#endif
        // all
        //!nope!:if (GetCVar("k8GoreOpt_BloodAmount") >= 0) A_CustomMissile("K8Gore_BloodHit$(Color)", 0, 0, random(0, 360), CMF_AIMDIRECTION, random(0, 90));
      }
      TNT1 AAAA 0 A_CustomMissile("K8Gore_FlyingBloodTrail$(Color)", 0, 0, random (0, 360), CMF_AIMDIRECTION, random (0, 180))
      /*TNT1 A 0 A_k8ConLog("blood: Spawn Done")*/
      TNT1 A 0 A_JumpIf(IsListenServer, "InListenServer")
      Stop

    InListenServer:
      TNT1 A 0 A_ChangeFlag("k8ServerSideOnly", false)
      /*TNT1 A 0 A_k8ConLog("blood: listen server")*/
      /* fallback */

    /* for both server modes if `SpawnInListenServer` is absent */
    SpawnInNetServer:
      TNT1 A 0
      /* for non-dedicated server we still want blood */
      /*TNT1 A 0 A_k8ConLog("blood: dedicated server")*/
      TNT1 A 8
      Stop

    SpawnInListenServer:
      TNT1 A 0
      TNT1 A 0 A_ChangeFlag("k8ServerSideOnly", true)
      /*TNT1 A 0 A_k8ConLog("blood: listen server spawn")*/
      Goto Spawn

    NetDetachedOnClient:
      TNT1 A 0
      /*TNT1 A 0 A_k8ConLog("blood: detached")*/
      Goto Spawn

    /*
    Spray:
      TNT1 A 0 A_SetUserVar("user_spray", 1)
      Goto Spawn
    */
  }
}

// flying blood; will spawn blood spots and ceiling blood
Actor K8Gore_FlyingBlood$(Color) /*: K8Gore_BloodBase$(TransientSfx)*/ {
  $(Translation)
  game Doom
  scale 0.5
  speed 4
  health 1
  radius 8
  height 2
  Gravity 0.4 //0.8
  damage 0
  DamageType Blood
  Decal K8GoreBloodSuper$(Color)
#ifdef TRANSIENT
  FlatDecal K8GoreBloodSuper$(Color)
#else
  FlatDecal K8GoreBloodSuperPers$(Color)
#endif
  +SeparateFlatDecal
  +Decals2SPass
  +MISSILE
  +CLIENTSIDEONLY
  +NOTELEPORT
  +NOBLOCKMAP
  +THRUACTORS
  +BLOODLESSIMPACT
  +FORCEXYBILLBOARD
  +NODAMAGETHRUST
  +MOVEWITHSECTOR
  +CORPSE
  -DONTSPLASH
  -NOGRAVITY
  +NoTeleport
  +CopyBloodTranslation
  //+KillOnUnarchive
  +NoTrigger
  +k8DetachFromServer
  +k8ServerSideOnly
  +k8AllowSimpleTick
  Mass 1

  States {
    Spawn:
      TNT1 A 0
      /*TNT1 A 0 A_k8ConLog("fake blood missile!")*/
      TNT1 A 0 {
        if (!GetCVar("k8GoreOpt_FlatDecals")) {
          A_SetFlatDecal("None");
        }
        ThrustThingZ(0, random(2, 4)*4, 0, 1);
      }
      //TNT1 A 0 ThrustThing(angle*256.0/360.0, random(0, 2), 1)
      //TNT1 A 0 ThrustThing(random(0, 255), random(0, 2), 1)
      B8OD AB 2
      TNT1 A 0 A_JumpIf(waterlevel > 1, "Splash")
      TNT1 A 0 A_Jump(256, "Spawn1", "Spawn2", "Spawn3")

    Spawn1:
      B8OD C 0
      Goto MainWorker
    Spawn2:
      B8OD D 0
      Goto MainWorker
    Spawn3:
      B8OD E 0
      Goto MainWorker

    MainWorker:
      "####" "#" 0 A_JumpIf(waterlevel > 1, "Splash")
    MainLoop:
#ifndef TRANSIENT
      "####" "#" 100
#else
      "####" "#" 1 A_FadeOut(0.01)
#endif
      Loop

    Splash:
      "####" "#" 1 A_FadeOut(0.01)
      Loop

    // hits something
    Death:
      TNT1 A 0
      //k8: don't make any sound, it is not noticeable anyway
      //TNT1 A 0 { if (GetCVar("k8GoreOpt_BloodDripSound") > 0) A_PlaySound_NoNet("k8gore/blooddrop2"); }
      //"####" "#" 0 A_JumpIf(waterlevel > 1, "DeathDone")
      //TNT1 A 0 A_SpawnItem("K8Gore_BloodSpot$(Color)", 0, 1, false, true)
      //TNT1 A 0 A_CustomMissile("K8Gore_CeilBloodChecker$(Color)", -15, 0, 90, CMF_AIMDIRECTION, 90)
      TNT1 A 0
        {
          if (waterlevel == 0) {
            // spawn floor blood spot
            if (GetCVar("k8GoreOpt_FloorBlood")) {
              if (!GetCVar("k8GoreOpt_FlatDecals") || !IsFlatDecalsAllowed()) {
                A_SpawnItem("K8Gore_BloodSpot$(Color)", 0, 1, false, true);
              } else if (IsActorOnALiquidFloor()) {
                A_SpawnItem("K8Gore_BloodSpot$(Color)", 0, 1, false, true);
              }
            }
            // spawn ceiling blood spot
            if (GetCVar("k8GoreOpt_CeilBlood") && GetCVar("k8GoreOpt_BloodAmount") >= 2) {
              if (ceilingz-z < random(118, 132)) {
                if ((ceilingz-z <= 64 && random(0, 100) > 20) ||
                    (ceilingz-z > 64 && ceilingz-z < 80 && random(0, 100) > 40) ||
                    (ceilingz-z >= 80 && random(0, 100) > 60))
                {
                  //A_k8ConLog(va("z=%s; cz=%s; czdist=%s", z, ceilingz, ceilingz-z));
                  A_SpawnItem("K8Gore_CeilBloodSpot$(Color)", 0/*distance*/, 0/*zheight*/, 0/*useammo*/, true/*transferblood*/);
                }
              }
            }
          }
        }
      Stop
    DeathDone:
      TNT1 A 1
      Stop
   }
}

// this is floor blood spot (it may be spawned in the air, though)
Actor K8Gore_BloodSpot$(Color) : K8Gore_BloodBase$(TransientSfx) {
  LimitWithSubCvar "k8GoreOpt_MaxBloodSplatEntities"
  $(Translation)
  game Doom
  radius 0
  height 0
  Health 300
  +NOTELEPORT
  +CLIENTSIDEONLY
  +MOVEWITHSECTOR
  +FLOORCLIP
  +NOTARGET
  -SHOOTABLE
  +NOBLOOD
  -DONTSPLASH
  -SOLID
  +THRUACTORS
  -NOGRAVITY
  -NoLiftDrop
  +NOBLOCKMAP
  +OrientedOfs
  +CopyBloodTranslation
  +k8DetachFromServer
  +k8ServerSideOnly
  +k8AllowSimpleTick
  Renderstyle Dark
#ifdef TRANSIENT
  Alpha 0.85
#else
  Alpha 0.9
#endif
  Mass 99999
  Scale 0.5
  Decal None
  -SeparateFlatDecal
  -Decals2SPass
  //Decal K8GoreBloodSplat$(Color)
  //+Decals2SPass

  States {
    Spawn:
      TNT1 A 0
      TNT1 A 0 {
        A_LayToFloorWithYawSprite(Random(0, 360));
        A_SetScaleXExt(Random(3, 9)/10.0);
      }
      TNT1 A 0 A_JumpIf(waterlevel > 1, "SplashWater")

    SpawnNormally:
      TNT1 A 0 A_Jump(256, "SpawnAFrameOnly", "SpawnBFrameOnly", "SpawnCFrameOnly", "SpawnDFrameOnly")
    SpawnAFrameOnly:
      BPDL A 0
      Goto DoAfterSpawn
    SpawnBFrameOnly:
      BPDL B 0
      Goto DoAfterSpawn
    SpawnCFrameOnly:
      BPDL C 0
      Goto DoAfterSpawn
    SpawnDFrameOnly:
      BPDL D 0
      Goto DoAfterSpawn

#ifdef TRANSIENT
    DoAfterSpawn:
      "####" "#" 35 A_CheckFloor("ItLanded")
      Loop
    ItLanded:
      "####" "#" 0
        {
          A_ChangeVelocity(0, 0, 0, CVF_REPLACE);
          A_ChangeFlag("NoInteraction", true);
          if (!A_k8JumpIfOnFloorSky("Death")) {
            A_k8JumpIfLiquidFloor("ItLandedOnWater");
          }
        }
      //"####" "#" random(100, 450)
      // become non-scripted
      "####" "#" -1 A_BecomeK8VavoomInternalNoTickGrav(frandom(100, 450)/35.0, frandom(64, 90)/35.0/10.0)
    FadeOutLoop:
      "####" "#" random(64, 90)
      "####" "#" 0 A_FadeOut(0.01)
      Loop
#else
    DoAfterSpawn:
      "####" "#" 16 A_CheckFloor("ItLanded")
      Loop
    ItLanded:
      "####" "#" 0
        {
          A_ChangeVelocity(0, 0, 0, CVF_REPLACE);
          A_ChangeFlag("NoInteraction", true);
          if (!A_k8JumpIfOnFloorSky("Death")) {
            A_k8JumpIfLiquidFloor("ItLandedOnWater");
          }
        }
    EndlessLoop:
      // become non-scripted; from 22 secs to 17 minutes
      "####" "#" -1 A_BecomeK8VavoomInternalNoTickGrav(frandom(800, 12600)/35.0, frandom(64, 90)/35.0/10.0)
      /*
      "####" "#" 800
      // it still may disappear after some time
      "####" "#" 0 A_Jump(16, "NormalFadeOut")
      */
      Loop
    NormalFadeOut:
      "####" "#" random(64, 90)
      "####" "#" 0 A_FadeOut(0.01)
      Loop
#endif

    SplashWater:
      TNT1 A 0 A_FadeOut(0.3)
      TNT1 A 0 A_Jump(256, "SpawnAFrameOnlyWater", "SpawnBFrameOnlyWater", "SpawnCFrameOnlyWater", "SpawnDFrameOnlyWater")
    SpawnAFrameOnlyWater:
      BPDL A 0
      Goto WaterFadeMain
    SpawnBFrameOnlyWater:
      BPDL B 0
      Goto WaterFadeMain
    SpawnCFrameOnlyWater:
      BPDL C 0
      Goto WaterFadeMain
    SpawnDFrameOnlyWater:
      BPDL D 0
      Goto WaterFadeMain

    WaterFadeMain:
      "####" "#" 16 A_CheckFloor("ItLandedOnWater")
      Loop
    ItLandedOnWater:
      "####" "#" 0
        {
          A_ChangeVelocity(0, 0, 0, CVF_REPLACE);
          A_ChangeFlag("NoInteraction", true);
        }
      // stay for some time
      "####" "#" random(50, 100)
      // and fade out
    WaterFadeLoop:
      "####" "#" random(2, 4)
      "####" "#" 0 A_FadeOut(0.01)
      Loop


    Death:
      TNT1 A 0 A_NoBlocking
      Stop
  }
}

  +NoTrigger

  //+Decals2SPass

  Alpha 0.99
  +k8ServerSideOnly
  +k8AllowSimpleTick

  States {
    Spawn:
      BPDL A 0
      "####" "#" 0 {
        if (!GetCVar("k8GoreOpt_FlatDecals")) {
          A_SetFlatDecal("None");
        }
      }
      // if flat decals are allowed, get lost
      "####" "#" 0 A_JumpIfCVarFalse("k8GoreOpt_FlatDecals", "GoOnPool")
      "####" "#" 0 A_JumpIf(IsFlatDecalsAllowed(), "Die")
    GoOnPool:
      BPDL A 0 A_Jump(256, "Pool1", "Pool2")

      BPDL A 0
      Goto MainProcessor

    Pool2:
      BPDL B 0
      Goto MainProcessor

    MainProcessor:
      "####" "#" 0 A_JumpIf(waterlevel > 1, "Splash")
      "####" "#" 0 A_LayToFloorWithYawSprite(Random(0, 360))
      "####" "#" 0 A_SetScaleXExt(Random(6, 12)/10.0)

    WaitFloor:
      "####" "#" 5 A_CheckFloor("ItLanded")
      Loop
    ItLanded:
      "####" "#" 0
        {
          A_ChangeVelocity(0, 0, 0, CVF_REPLACE);
          A_ChangeFlag("NoInteraction", true);
          if (!A_k8JumpIfOnFloorSky("Die")) {
            A_k8JumpIfLiquidFloor("ItLandedOnWater");
          }
        }

    Live:
      "####" "#" random(400, 900)
      "####" "#" 0 A_Jump(16, "ItLandedOnWater") // fadeout
      Loop

    ItLandedOnWater:
      "####" "#" 5 A_FadeOut(0.1)
      Loop

    Splash:
      B8OD A 0
      Stop

    Death:
      "####" "#" 0
        {
          A_ChangeVelocity(0, 0, 0, CVF_REPLACE);
          A_ChangeFlag("NoInteraction", true);
        }
      Goto WaitFloor

    Die:
      TNT1 A 1
      Stop
  }
}
