// flat bootprints -----------------------------------------------------------
// you can override bootprints
// overriden bootprint will delete the old one, unless you use
//   bootprint Name modify { ... }
// the following are modified in Doom/Hexen/etc. definitions

// this is used to create shaded decals
// time doesn't matter
bootprint K8Gore_BootPrintDecal {
  decal  K8Gore_BootPrintDecal
  time   default  // or: time minsecflt maxsecflt
  basedecal BaseBootPrint {
    // do not change shade!
    shade "50 00 00"
    animator K8GoreBloodPrint_FaderAnim
  }
}

// this is used to create shaded decals
// time doesn't matter
bootprint K8Gore_BootPrintDecalTransient {
  decal  K8Gore_BootPrintDecalTransient
  time   default  // or: time minsecflt maxsecflt
  basedecal BaseBootPrint {
    // do not change shade!
    shade "4f 00 00"
    animator K8GoreBloodPrint_FaderAnimTransient
  }
}


bootprint VanillaWater {
  decal  WaterLiquidBootPrint
  time   default  // or: time minsecflt maxsecflt
  // if there is no decal named "WaterLiquidBootPrint",
  // it will be created from base decal
  // you can change:
  //   scale
  //   translucent
  //   shade
  basedecal BaseBootPrint {
    shade "00 00 4f"
    translucent 0.6
    animator LiquidBootPrintFader
  }
}
